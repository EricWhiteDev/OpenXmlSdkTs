        openXml.util.isBase64 = function (str) {
            if (typeof str !== 'string') {
                return false;
            }
            var sub = str.substring(0, 500);
            for (var i = 0; i < sub.length; i++) {
                var s = sub[i];
                if (s >= 'A' && s <= 'Z') continue;
                if (s >= 'a' && s <= 'z') continue;
                if (s >= '0' && s <= '9') continue;
                if (s === '+' || s === '/') continue;
                return false;
            }
            return true;
        }
